``` js 
var x = 1;
const y = 2; 

// 함수 정의 
function foo(a) {
	// 지역 변수 선언
```

전역 변수 선언에 var, const 가 사용됨. 
var, const 모두 -> 영향을 미치는 범위를 보면, 가장 하위에 있는 스코프에 까지 영향을 미침. 

근데, 어떤 코드에서 보면, 
let 은 { } 블록을 넘어서 영향을 못 미치던데? 
let 으로 사용되면 -> 같은 이름이더라도, '별개의 변수'⭐⭐⭐⭐⭐  p211 가 된다. 
``` js
let foo = 1; 

{
	let foo = 2;
	let foo = 3; 
}

```


``` js
var foo = 1; 

{
	var foo = 2;
	var foo = 3; 
}

console.log(foo)  // 2
```
